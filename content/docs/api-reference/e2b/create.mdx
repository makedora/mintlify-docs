---
title: Create Sandbox
description: Create a new E2B sandbox
---

<ParamField body="name" type="string" required>Sandbox name.</ParamField>
<ParamField body="taskDescription" type="string">Description used to generate a display name via AI.</ParamField>
<ParamField body="template" type="string" default="yolocode">Template: `yolocode`, `yolocode-dev`, or `default`.</ParamField>
<ParamField body="gitHubRepo" type="string">GitHub repo to clone (format: `owner/repo`).</ParamField>
<ParamField body="githubToken" type="string">GitHub token for repo access.</ParamField>
<ParamField body="githubBranch" type="string">Branch to checkout.</ParamField>
<ParamField body="claudeCredentials" type="object">Claude OAuth credentials to inject.</ParamField>
<ParamField body="repositorySetup" type="object">Setup config with `environmentVariables` map.</ParamField>
<ParamField body="metadata" type="object">Additional metadata key-value pairs.</ParamField>

### Response

<ResponseField name="id" type="string">Sandbox ID.</ResponseField>
<ResponseField name="name" type="string">Sandbox name.</ResponseField>
<ResponseField name="displayName" type="string">AI-generated display name.</ResponseField>
<ResponseField name="status" type="string">`running`</ResponseField>
<ResponseField name="host" type="string">Sandbox host URL.</ResponseField>
<ResponseField name="websshHost" type="string">WebSSH URL (port 8888).</ResponseField>
<ResponseField name="claudeHost" type="string">Claude WebSocket URL (port 9999).</ResponseField>
