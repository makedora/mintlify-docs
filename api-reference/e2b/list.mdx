---
title: List Sandboxes
api: GET /api/e2b-sandboxes
description: List E2B sandboxes
---

<ParamField query="state" type="string" default="all">
  Filter by state: `running`, `paused`, or `all`.
</ParamField>

### Response
Returns an array of sandbox objects filtered to the authenticated user.

<ResponseField name="sandboxes" type="array">
  <Expandable title="Sandbox object">
    <ResponseField name="id" type="string">Sandbox ID.</ResponseField>
    <ResponseField name="name" type="string">Sandbox name.</ResponseField>
    <ResponseField name="displayName" type="string">AI-generated human-readable name.</ResponseField>
    <ResponseField name="status" type="string">Current status: `running` or `stopped`.</ResponseField>
    <ResponseField name="createdAt" type="string">ISO 8601 timestamp.</ResponseField>
    <ResponseField name="host" type="string">Sandbox host URL.</ResponseField>
  </Expandable>
</ResponseField>
